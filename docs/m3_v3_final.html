"""
DistilBERT medical query classifier.
Written in M2 Task 3 by Doha. This shows the interface src/pipeline.py expects.
If your M2 implementation uses different method names, update to match.

src/pipeline.py calls:
    from src.classification.classifier import MedicalClassifier
    _clf = MedicalClassifier.load(str(settings.classifier_path))
    category = _clf.predict(question_text)
"""
from __future__ import annotations
from config.settings import settings


class MedicalClassifier:
    CATEGORIES = settings.categories  # ["Symptoms","Diagnosis","Treatment","Medication","Prevention","General"]

    def __init__(self, model, tokenizer):
        self.model = model
        self.tokenizer = tokenizer

    @classmethod
    def load(cls, model_path: str) -> "MedicalClassifier":
        """Load fine-tuned DistilBERT from models/classifier/ directory."""
        from transformers import AutoTokenizer, AutoModelForSequenceClassification
        tokenizer = AutoTokenizer.from_pretrained(model_path)
        model = AutoModelForSequenceClassification.from_pretrained(model_path)
        model.eval()
        return cls(model=model, tokenizer=tokenizer)

    def predict(self, text: str) -> str:
        """Classify text into one of 6 medical categories. Returns category string."""
        import torch
        inputs = self.tokenizer(text, return_tensors="pt", truncation=True, max_length=512)
        with torch.no_grad():
            idx = self.model(**inputs).logits.argmax(dim=-1).item()
        return self.CATEGORIES[idx] if 0 <= idx < len(self.CATEGORIES) else "General"
src/rag/vectorstore.py — Required interface
src/rag/vectorstore.py — interface (M2 T1, Owner: Ziad)python
"""
FAISS vector store for PubMedQA embeddings.
Written in M2 Task 1 by Ziad. This shows the interface src/pipeline.py expects.

src/pipeline.py calls:
    from src.rag.vectorstore import FAISSVectorStore
    _vs = FAISSVectorStore.load(str(settings.faiss_index_path))
    results = _vs.search(question, k=5)
    # results: List[Tuple[chunk_id_str, chunk_text_str]]
"""
from __future__ import annotations
from typing import List, Tuple
from pathlib import Path
import pickle


class FAISSVectorStore:
    def __init__(self, index, chunk_mapping: dict, embedding_model):
        self.index = index
        self.chunk_mapping = chunk_mapping
        self.embedding_model = embedding_model

    @classmethod
    def load(cls, index_dir: str) -> "FAISSVectorStore":
        """
        Load FAISS index from data/embeddings/faiss_index/ directory.
        Directory must contain:
          - pubmedqa_index_flatl2.faiss
          - chunk_mapping.pkl   (dict: {int_id -> (chunk_id_str, chunk_text_str)})
        """
        import faiss
        from sentence_transformers import SentenceTransformer
        from config.settings import settings

        p = Path(index_dir)
        index = faiss.read_index(str(p / "pubmedqa_index_flatl2.faiss"))
        with open(p / "chunk_mapping.pkl", "rb") as f:
            chunk_mapping = pickle.load(f)
        embedding_model = SentenceTransformer(settings.embedding_model)
        return cls(index=index, chunk_mapping=chunk_mapping, embedding_model=embedding_model)

    def search(self, query: str, k: int = 5) -> List[Tuple[str, str]]:
        """
        Search for top-k relevant chunks.
        Returns: List of (chunk_id: str, chunk_text: str) tuples, ordered by relevance.
        """
        q_emb = self.embedding_model.encode([query], convert_to_numpy=True, normalize_embeddings=True)
        _, indices = self.index.search(q_emb, k)
        results = []
        for idx in indices[0]:
            if idx == -1 or idx not in self.chunk_mapping:
                continue
            item = self.chunk_mapping[idx]
            if isinstance(item, tuple):
                results.append(item)
            else:
                results.append((ff"chunk_{idx:04d}", str(item)))
        return results
src/rag/pipeline.py — Required interface
src/rag/pipeline.py — interface (M2 T2, Owner: Youssef)python
"""
LangChain RAG generation pipeline.
Written in M2 Task 2 by Youssef. This shows the interface src/pipeline.py expects.

src/pipeline.py calls:
    from src.rag.pipeline import build_rag_chain
    _chain = build_rag_chain(vectorstore, top_k, llm_model, max_new_tokens)
    answer = _chain.generate(question, context_list, category)

NOTE: The disclaimer is NOT added here — it is added by api/routes/query.py.
"""
from __future__ import annotations
from typing import List, Any
import types


class RAGChain:
    def __init__(self, llm, top_k: int, max_new_tokens: int):
        self.llm = llm
        self.top_k = top_k
        self.max_new_tokens = max_new_tokens

    def generate(self, question: str, context: List[str], category: str) -> str:
        """
        Generate answer from question + retrieved context chunks.
        Args:
            question: original medical question string
            context:  list of retrieved chunk text strings (from FAISS)
            category: predicted category string (from classifier)
        Returns:
            str — generated answer text. No disclaimer added here.
        """
        context_text = "\n\n".join(context[:5])
        prompt = (
            ff"You are a medical information assistant. Answer this {category} question "
            ff"based only on the provided context.\n\nContext:\n{context_text}\n\n"
            ff"Question: {question}\n\nAnswer:"
        )
        inputs = self.llm.tokenizer(prompt, return_tensors="pt", max_length=512, truncation=True)
        outputs = self.llm.model.generate(**inputs, max_new_tokens=self.max_new_tokens, do_sample=False)
        return self.llm.tokenizer.decode(outputs[0], skip_special_tokens=True).strip()


def build_rag_chain(vectorstore: Any, top_k: int = 5,
                    llm_model: str = "google/flan-t5-base",
                    max_new_tokens: int = 256) -> RAGChain:
    """Build and return a RAGChain. Called once at API startup."""
    from transformers import AutoTokenizer, AutoModelForSeq2SeqLM
    tokenizer = AutoTokenizer.from_pretrained(llm_model)
    model = AutoModelForSeq2SeqLM.from_pretrained(llm_model)
    model.eval()
    llm = types.SimpleNamespace(tokenizer=tokenizer, model=model)
    return RAGChain(llm=llm, top_k=top_k, max_new_tokens=max_new_tokens)

"""
Integrated M2 pipeline — the single function the API calls.
Built in M2 Task 5 by Abdelrahman.

Returns exactly: {"answer": str, "category": str, "sources": List[str]}
The disclaimer is NOT here — it is injected by api/routes/query.py.

Models are loaded lazily (once, on first request) to avoid slow startup.
"""
from __future__ import annotations
from typing import Dict, Any
import logging
from config.settings import settings

logger = logging.getLogger("pipeline")

# Module-level cache — loaded once, reused on every request
_classifier  = None
_rag_chain   = None
_vectorstore = None


def _load_models():
    """Load all ML models on first call. Thread-safe via Python's GIL."""
    global _classifier, _rag_chain, _vectorstore
    if _classifier is not None:
        return

    logger.info("Loading models — this takes 10-30s on first request...")

    from src.classification.classifier import MedicalClassifier
    _classifier = MedicalClassifier.load(str(settings.classifier_path))
    logger.info("✓ Classifier loaded")

    from src.rag.vectorstore import FAISSVectorStore
    _vectorstore = FAISSVectorStore.load(str(settings.faiss_index_path))
    logger.info("✓ FAISS index loaded")

    from src.rag.pipeline import build_rag_chain
    _rag_chain = build_rag_chain(
        vectorstore=_vectorstore,
        top_k=settings.retrieval_top_k,
        llm_model=settings.llm_model,
        max_new_tokens=settings.max_new_tokens
    )
    logger.info("✓ RAG chain ready")


def run_pipeline(question: str) -> Dict[str, Any]:
    """
    Run the full integrated pipeline on a medical question.

    Args:
        question: plain text medical question from the API request

    Returns:
        {"answer": str, "category": str, "sources": List[str]}
        — the API layer adds the disclaimer on top of this

    Called on every POST /query request.
    """
    _load_models()

    category = _classifier.predict(question)
    if category not in settings.categories:
        category = "General"

    retrieved = _vectorstore.search(question, k=settings.retrieval_top_k)
    source_ids    = [r[0] for r in retrieved]
    context_texts = [r[1] for r in retrieved]

    answer = _rag_chain.generate(question=question, context=context_texts, category=category)

    return {
        "answer":   answer,
        "category": category,
        "sources":  source_ids,
    }
scripts/download_models.py — For teams using Azure Blob storage for weights
scripts/download_models.pypython
"""
Download model weights from Azure Blob Storage.
Only needed if weights exceed GitHub's 100MB limit.

Usage:
    python scripts/download_models.py        # downloads if needed
    # OR add to Dockerfile: RUN python scripts/download_models.py

If AZURE_STORAGE_CONNECTION is not set, prints a message and exits cleanly.
"""
import os, logging, sys
from pathlib import Path

logging.basicConfig(level=logging.INFO, format="%(asctime)s | %(message)s")
logger = logging.getLogger("download_models")

FILES = {
    "classifier/config.json":             "models/classifier/config.json",
    "classifier/pytorch_model.bin":        "models/classifier/pytorch_model.bin",
    "classifier/tokenizer_config.json":    "models/classifier/tokenizer_config.json",
    "classifier/vocab.txt":                "models/classifier/vocab.txt",
    "faiss_index/pubmedqa_index_flatl2.faiss": "data/embeddings/faiss_index/pubmedqa_index_flatl2.faiss",
    "faiss_index/chunk_mapping.pkl":       "data/embeddings/faiss_index/chunk_mapping.pkl",
}

def main():
    conn = os.getenv("AZURE_STORAGE_CONNECTION")
    if not conn:
        logger.info("AZURE_STORAGE_CONNECTION not set — using Git LFS or local files")
        return
    try:
        from azure.storage.blob import BlobServiceClient
    except ImportError:
        logger.error("pip install azure-storage-blob"); sys.exit(1)

    container = os.getenv("AZURE_STORAGE_CONTAINER", "model-weights")
    cc = BlobServiceClient.from_connection_string(conn).get_container_client(container)

    for blob_name, local_path in FILES.items():
        lp = Path(local_path)
        if lp.exists():
            logger.info(ff"Already exists: {local_path}"); continue
        lp.parent.mkdir(parents=True, exist_ok=True)
        logger.info(ff"Downloading: {blob_name}")
        with open(lp, "wb") as f:
            f.write(cc.get_blob_client(blob_name).download_blob().readall())
        logger.info(ff"✓ {local_path} ({lp.stat().st_size/1024/1024:.1f}MB)")
    logger.info("✓ All model files ready")

if __name__ == "__main__":
    main()

from pydantic import BaseModel, Field
from typing import List, Optional


class QueryRequest(BaseModel):
    question: str = Field(..., min_length=5, max_length=1000,

@app.middleware("http")
async def latency_middleware(request: Request, call_next):
    """Logs latency of every request. Injects X-Response-Time-Ms header."""
    start = time.perf_counter()
    response = await call_next(request)
    ms = round((time.perf_counter() - start) * 1000, 2)
    response.headers["X-Response-Time-Ms"] = str(ms)
    logger.info(ff"{request.method} {request.url.path} → {response.status_code} | {ms}ms")
    return response

@app.exception_handler(Exception)
async def global_exception_handler(request: Request, exc: Exception):
    logger.error(ff"Unhandled: {exc}", exc_info=True)
    return JSONResponse(status_code=500, content={"error": str(exc)})

def warm_up():
    print(f"API: {BASE_URL}")
    print("Warm-up request (not recorded)...")
    try:
        r = requests.get(ff"{BASE_URL}/health", timeout=120)
        print(ff"  Warm-up: HTTP {r.status_code} ({r.elapsed.total_seconds():.1f}s)")
    except Exception as e:
        print(ff"  Warm-up failed: {e}. Wait 90s and retry.")


def run_query(q: str, n: int) -> dict:
    try:
        start = time.perf_counter()
        r = requests.post(ff"{BASE_URL}/query", json={"question": q}, timeout=30)
        ms = round((time.perf_counter() - start) * 1000, 1)
        if r.status_code == 200:
            d = r.json()
            return {"n":n,"q":q,"ms":ms,"cat":d.get("category","MISS"),
                    "disc_ok":bool(d.get("disclaimer")),
                    "lat_pass":ms<=LIMIT_MS,
                    "cat_valid":d.get("category") in VALID_CATS,"err":None}
        return {"n":n,"q":q,"ms":ms,"cat":"ERR","disc_ok":False,
                "lat_pass":False,"cat_valid":False,"err":ff"HTTP {r.status_code}"}
    except requests.exceptions.Timeout:
        return {"n":n,"q":q,"ms":30000,"cat":"TIMEOUT","disc_ok":False,
                "lat_pass":False,"cat_valid":False,"err":"Timeout 30s"}


if __name__ == "__main__":
    warm_up()
    print("Waiting 5s for warm instance..."); time.sleep(5)
    results = []
    for i, (q, exp) in enumerate(TEST_QUERIES, 1):
        r = run_query(q, i); r["expected"] = exp; results.append(r)
        ok = "✅" if r["lat_pass"] and r["disc_ok"] else "❌"

    lat_pass  = sum(1 for r in results if r["lat_pass"])

Latency ≤5000ms:  {lat_pass}/{len(results)}  {'✅' if lat_pass==len(results) else '❌'}
Disclaimer OK:    {disc_pass}/{len(results)}  {'✅' if disc_pass==len(results) else '❌'}
Avg latency:      {avg_ms:.0f}ms
Max latency:      {max(r['ms'] for r in results):.0f}ms
Test date:        {datetime.now().strftime('%Y-%m-%d %H:%M')}
{'='*52}""")

    os.makedirs("reports", exist_ok=True)

        writer = csv.DictWriter(f, fieldnames=["n","q","expected","cat","ms","disc_ok","lat_pass","cat_valid","err"])
        writer.writeheader(); writer.writerows(results)
    print("Exported: reports/latency_results.csv")

```
  → Latency middleware (logs X-Response-Time-Ms header)
  → src/pipeline.run_pipeline()
      → DistilBERT classify (src/classification/classifier.py)
      → FAISS retrieve top-5 (src/rag/vectorstore.py)
      → flan-t5-base generate (src/rag/pipeline.py)
      → disclaimer injected from config.settings.disclaimer
  → JSON response: {answer, category, retrieved_sources, disclaimer}
  → Docker image (docker/Dockerfile)
  → Azure Container Registry (healthcareragacr.azurecr.io)
  → Azure App Service F1 (healthcare-rag-app.azurewebsites.net)
```
